using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Lista_1
{
    class Program

        static void Main(string[] args)
        {
            List<int> A = new List<int>();
            List<int> B = new List<int>();

            Console.WriteLine("Informe 10 n√∫meros");

            }
            Console.ReadLine();
